\begin{thm}[Mean Value Theorem]\label{thm:mean_value_theorem}
   \begin{gather*}
     \forall a, b \in \mathbb{R} \\
     f: [a, b] \to \mathbb{R} \text{ differentiable in } (a, b) \implies \\
     \exists c \in (a, b) \quad f'(c) = \frac{f(b) - f(a)}{b - a} 
   \end{gather*}
\end{thm}
